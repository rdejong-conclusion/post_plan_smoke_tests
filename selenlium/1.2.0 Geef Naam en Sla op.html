1.2.0 Geef Naam en Sla op
open /HARM/controller?action=bbg_algemeen_show
storeEval new Date().getFullYear() Jaar
storeEval new Date().getMonth() Maand
storeEval new Date().getDate() Dag
storeEval new Date().getHours() Uur
echo ${Jaar}-${Maand}-${Dag}
clickAndWait link=Algemeen
type name=bbg_config_naam Nederland-${Jaar}-${Maand}-${Dag} ${Uur}
clickAndWait id=button_sla_op
